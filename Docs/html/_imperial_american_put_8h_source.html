82CNicolsonPricer::CNicolsonPricer(double S_now, double K, double T, double Vol, double r, int I, double delta_t)
83{
84 int i, j, k;
85 double error, diff;
86 const double MAXk = 100;
87 const double TOL = 0.00000001;
88 const double DELTA_S = 3 * K / I;
89 const int i_STAR = floor(S_now / DELTA_S);
90 const double WEIGHT = (S_now - i_STAR * DELTA_S) / DELTA_S;
91 const double J = floor(T / delta_t);
92 delta_t = T / J;
93 const double W = 1.5;
94
95 double* predictor = new double[I + 1];
96 double* Vprevious_j = new double[I + 1];
97 double* Vcurrent_j = new double[I + 1];
98 double* fix = new double[I + 1];
99 double* a = new double[I + 1];
100 double* b = new double[I + 1];
101 double* c = new double[I + 1];
102 double* A = new double[I + 1];
103 double* B = new double[I + 1];
104 double* C = new double[I + 1];
105
106 for (i = 0; i <= I; i++)
107 {
108 Vprevious_j[i] = maxx(0, K - i * DELTA_S);
109 fix[i] = Vprevious_j[i];
110 }
111
112 for (i = 1; i <= I - 1; i++)
113 {
114 a[i] = delta_t / 4 * (Vol * Vol * i * i - r * i);
115 b[i] = (1 - delta_t / 2 * (r + Vol * Vol * i * i));
116 c[i] = (delta_t / 4 * (Vol * Vol * i * i + r * i));
117 A[i] = -a[i];
118 B[i] = (1 + delta_t / 2 * (r + Vol * Vol * i * i));
119 C[i] = -c[i];
120 }
121
122 Vcurrent_j[0] = K;
123 Vcurrent_j[I] = 0;
124
125 for (j = 1; j <= J; j++) {
126 for (i = 1; i <= I - 1; i++)
127 {
128 Vcurrent_j[i] = Vprevious_j[i];
129 }
130 for (i = 1; i <= I - 1; i++)
131 {
132 predictor[i] = a[i] * Vprevious_j[i - 1] + b[i] * Vprevious_j[i] + c[i] * Vprevious_j[i + 1];
133 }
134 k = 0;
135 do {
136 error = 0;
137 for (i = 1; i <= I - 1; i++)
138 {
139 diff = (predictor[i] - A[i] * Vcurrent_j[i - 1] - C[i] * Vcurrent_j[i + 1]) / B[i] - Vcurrent_j[i];
140 error += diff * diff;
141 Vcurrent_j[i] += W * diff;
142 }
143 k++;
144 } while (error > TOL && k < MAXk);
145 for (i = 1; i <= I - 1; i++)
146 Vcurrent_j[i] = maxx(Vcurrent_j[i], fix[i]);
147 for (i = 1; i <= I - 1; i++)
148 {
149 Vprevious_j[i] = Vcurrent_j[i];
150 }
151 }
152
153 Price = (1 - WEIGHT) * Vprevious_j[i_STAR] + WEIGHT * Vprevious_j[i_STAR + 1];
154
155 delete[] Vprevious_j;
156 delete[] Vcurrent_j;
157 delete[] predictor;
158 delete[] a;
159 delete[] b;
160 delete[] c;
161 delete[] A;
162 delete[] B;
163 delete[] C;
164}
